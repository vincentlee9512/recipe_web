    - category_2: STR (分类_2)(blank=True)
    - category_3: STR (分类_3)(blank=True)

    - step_2: TEXT
    - step_2_photo: STR (路径)
    - step_3: TEXT
    - step_3_photo: STR (路径)

    - step_8_photo: STR (路径)
    - step_9: TEXT
    - step_9_photo: STR (路径)
    - step_10: TEXT

    - is_published: BOOL
